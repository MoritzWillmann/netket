class PureRBM(nn.Module):
    """
    Encodes the pure state |Ïˆ><Ïˆ|, because it acts on row and column
    indices with the same RBM.
    """

    """The dtype of the weights."""

    """The nonlinear activation function."""
    alpha: Union[float, int] = 1

    """if True uses a bias in the dense layer (hidden layer bias)."""

    kernel_init: NNInitFunc = default_kernel_init
    """Initializer for the Dense layer matrix."""
    hidden_bias_init: NNInitFunc = zeros
    """Initializer for the hidden bias."""
    visible_bias_init: NNInitFunc = zeros
    """Initializer for the visible bias."""

    @nn.compact
    def __call__(self, Ïƒr, Ïƒc, symmetric=True):
        W = nknn.Dense(
            name="Dense",
            features=int(self.alpha * Ïƒr.shape[-1]),

            precision=self.precision,

            name="ASymm",
            features=int(self.alpha * Ïƒr.shape[-1]),
